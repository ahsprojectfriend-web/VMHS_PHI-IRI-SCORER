    let sessionResults = [], queue = [], activeGrade, currentQIdx, currentScore;
    let totalTestsInSequence = 1;
    let finalPayload = {};

    function autoStartExam() {
        const lrn = document.getElementById('lrn').value;
        if(lrn.length < 10) { alert("Please enter a valid LRN"); return; }
        activeGrade = parseInt(document.getElementById('studentGrade').value);
        document.getElementById('setup-page').style.display='none';
        runGradeSequence(activeGrade);
    }

    function runGradeSequence(level) {
        activeGrade = level; currentQIdx = 0; currentScore = 0;
        document.getElementById('transition-step').style.display = 'none';
        document.getElementById('test-step').style.display = 'block';
        document.getElementById('step-indicator').style.display = 'block';
        updateStepCounter();
        loadQuestion();
    }

    function updateStepCounter() {
        document.getElementById('step-indicator').innerText = `Assessment ${sessionResults.length + 1} of ${totalTestsInSequence}`;
    }

    function loadQuestion() {
        const exam = examData[activeGrade]; const q = exam.questions[currentQIdx];
        document.getElementById('part-label').innerText = `Grade ${activeGrade} Reading Material`;
        document.getElementById('passage-reference').innerText = exam.text;
        document.getElementById('quiz-area').innerHTML = `
            <div class="question-card">
                <div class="q-text">Q${currentQIdx + 1}: ${q.q}</div>
                <div class="options-grid">
                    ${q.options.map((opt, i) => `<button class="option-btn" onclick="handleChoice(${i})">${opt}</button>`).join('')}
                </div>
            </div>`;
        document.getElementById('passage-reference').scrollTop = 0;
    }

    function handleChoice(idx) {
        if(idx === examData[activeGrade].questions[currentQIdx].correct) currentScore++;
        currentQIdx++;
        if(currentQIdx < examData[activeGrade].questions.length) loadQuestion();
        else processResults();
    }

    function processResults() {
        const total = examData[activeGrade].questions.length;
        const pct = (currentScore / total);
        const profile = (pct >= 0.8) ? "Independent" : (pct >= 0.59) ? "Instructional" : "Frustration";

        sessionResults.push({ grade: activeGrade, score: `${currentScore}/${total}`, profile: profile });

        if (sessionResults.length === 1) {
            if (profile === "Frustration") {
                if (activeGrade - 3 >= 4) queue.push(activeGrade - 3);
                if (activeGrade - 2 >= 4) queue.push(activeGrade - 2);
                if (activeGrade - 1 >= 4) queue.push(activeGrade - 1);
            } else if (profile === "Instructional") {
                if (activeGrade - 1 >= 4) queue.push(activeGrade - 1);
                if (activeGrade + 1 <= 10) queue.push(activeGrade + 1);
            } else if (profile === "Independent") {
                if (activeGrade + 1 <= 10) queue.push(activeGrade + 1);
                if (activeGrade + 2 <= 10) queue.push(activeGrade + 2);
            }
            totalTestsInSequence = 1 + queue.length;
        }

        if (queue.length > 0) {
            const nextG = queue.shift();
            showTransition(nextG);
        } else {
            showFinalReport();
        }
    }

    function showTransition(nextG) {
        document.getElementById('test-step').style.display = 'none';
        document.getElementById('transition-step').style.display = 'block';
        document.getElementById('transition-msg').innerText = `Level Adjusting: Preparing Grade ${nextG} passage...`;
        setTimeout(() => runGradeSequence(nextG), 2500);
    }

    function showFinalReport() {
        document.getElementById('test-step').style.display = 'none';
        document.getElementById('results-step').style.display = 'block';
        
        let html = "";
        finalPayload = {
            lrn: document.getElementById('lrn').value,
            surname: document.getElementById('surname').value,
            firstName: document.getElementById('firstName').value,
            mi: document.getElementById('mi').value,
            enrollmentGrade: document.getElementById('studentGrade').value,
            section: document.getElementById('section').value
        };

        sessionResults.forEach(r => {
            html += `<div class="summary-card">
                        <h3 style="color:var(--vmhs-blue); margin-top:0;">Grade ${r.grade} Analysis</h3>
                        <p style="font-weight:bold; color:var(--vmhs-blue); font-size:1.2rem;">${r.profile}</p>
                        <h1 style="margin:10px 0; font-size:3rem;">${r.score}</h1>
                     </div>`;
            finalPayload['g' + r.grade] = r.profile;
        });

        document.getElementById('cumulative-results').innerHTML = html;
    }

    async function sendToDatabase() {
        const btn = document.getElementById('db-btn');
        btn.innerText = "UPLOADING TO DATABASE...";
        btn.disabled = true;

        const formData = new URLSearchParams();
        for (let key in finalPayload) {
            formData.append(key, finalPayload[key]);
        }

        try {
            await fetch(FINAL_GS_LINK, {
                method: 'POST',
                mode: 'no-cors', 
                body: formData.toString(),
                headers: { "Content-Type": "application/x-www-form-urlencoded" }
            });
            btn.innerText = "SUCCESS! RETURNING TO HOME...";
            btn.style.background = "#6c757d";
            setTimeout(() => location.reload(), 3000);
        } catch (e) {
            alert("Connection error. Please check internet and try again.");
            btn.innerText = "RETRY SENDING";
            btn.disabled = false;
        }
    }
